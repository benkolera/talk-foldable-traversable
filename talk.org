* Foldable

Something that has foldr (or foldMap)

#+BEGIN_SRC haskell
fold ["ab","c"]

instance Foldable [] where
  foldr :: (a -> b -> b) -> b -> [a] -> b
  foldr f z []     = z
  foldr f z (x:xs) = x `f` foldr f z xs
